00065             parent = &dummy;
00066         else if (sst[1]->s[0] == CCN_UNINIT) {
00067             parent = (struct strategy_state *)sst[1]->s;
00068             *parent = dummy;
00069         }
00070         else if ((sst[1]->s[0] & CCN_MAGIC_MASK) == MINE)

00076             if ((npe->magic & CCN_MAGIC_MASK) != MINE)
00077                 *npe = *parent;
00078             else

00080         }
00081         if (npe->magic != MINE)

00085         case CCNST_NOP:
00086             break;

00088             best = npe->src;

00092             for (x = strategy->pfl; x != NULL; x = x->next)
00093                 if ((x->pfi_flags & CCND_PFI_DNSTREAM) != 0)
00094                     break;
00095             if (x == NULL || (x->pfi_flags & CCND_PFI_PENDING) == 0)
00096                 return;
00097             if (best == CCN_NOFACEID) {
00098                 randlow = 4000;
00099                 randrange = 75000;
00100             }
00101             else {
00102                 randlow = npe->usec;
00103                 if (randlow < 2000)
00104                     randlow = 100 + ccnd_random(h) % 4096U;
00105                 randrange = (randlow + 1) / 2;
00106             }
00107             nleft = 0;
00108             for (p = strategy->pfl; p!= NULL; p = p->next) {
00109                 if ((p->pfi_flags & CCND_PFI_UPSTREAM) != 0) {
00110                     if (p->faceid == best) {
00111                         /* we may have already sent in case of TAP */
00112                         if ((p->pfi_flags & CCND_PFI_UPENDING) == 0)
00113                             p = send_interest(h, strategy->ie, x, p);
00114                         strategy_settimer(h, strategy->ie, npe->usec, CCNST_TIMER);
00115                     }
00116                     else if ((p->pfi_flags & CCND_PFI_UPENDING) != 0)
00117                         /* TAP interest has already been sent */;
00118                     else if (p->faceid == npe->osrc)
00119                         pfi_set_expiry_from_micros(h, strategy->ie, p, randlow);
00120                     else {
00121                         /* Want to preserve the order of the rest */
00122                         nleft++;
00123                         p->pfi_flags |= CCND_PFI_SENDUPST;
00124                     }
00125                 }
00126             }
00127             if (nleft > 0) {
00128                 /* Send remainder in order, with randomized timing */
00129                 amt = (2 * randrange + nleft - 1) / nleft;
00130                 if (amt == 0) amt = 1; /* paranoia - should never happen */
00131                 usec = randlow;
00132                 for (p = strategy->pfl; p!= NULL; p = p->next) {
00133                     if ((p->pfi_flags & CCND_PFI_SENDUPST) != 0) {
00134                         pfi_set_expiry_from_micros(h, strategy->ie, p, usec);
00135                         usec += ccnd_random(h) % amt;
00136                     }
00137                 }
00138             }
00139             break;
00140         case CCNST_UPDATE:
00141             /* Just go ahead and send as prompted */
00142             for (p = strategy->pfl; p!= NULL; p = p->next) {
00143                 if ((p->pfi_flags & CCND_PFI_ATTENTION) != 0) {
00144                     p->pfi_flags &= ~CCND_PFI_ATTENTION;
00145                     p->pfi_flags |= CCND_PFI_SENDUPST;
00146                 }
00147             }
00148             break;
00149         case CCNST_TIMER:
00150             /*
00151              * Our best choice has not responded in time.
00152              * Increase the predicted response.
00153              */
00154             adjust_predicted_response(h, npe, CCN_NOFACEID);
00155             adjust_predicted_response(h, parent, CCN_NOFACEID);
00156             break;
00157         case CCNST_SATISFIED:
00158             /* Keep a little history about where matching content comes from. */
00159             adjust_predicted_response(h, npe, faceid);
00160             adjust_predicted_response(h, parent, faceid);
00161             break;
00162         case CCNST_TIMEOUT:
00163             /* Interest has not been satisfied or refreshed */
00164             break;
00165         case CCNST_INIT:
00166             break; /* No strategy private data needed */
00167         case CCNST_EXPUP:
00168             break;
00169         case CCNST_EXPDN:
00170             break;
00171         case CCNST_REFRESH:
00172             break;
00173         case CCNST_FINALIZE:
00174             break; /* Nothing to clean up */
00175     }

00181 /**
00182  * Adjust the predicted response associated with a name prefix entry.
00183  *
00184  * It is decreased by a small fraction if we get content within our
00185  * previous predicted value, and increased by a larger fraction if not.
00186  *
00187  * faceid is CCN_NOFACEID if no content arrived, or else tells the
00188  * arrival face.
00189  */
00190 static void
00191 adjust_predicted_response(struct ccnd_handle *h,
00192                           struct strategy_state *s, unsigned faceid)
00193 {
00194     unsigned t = s->usec;
00195     if (faceid == CCN_NOFACEID) {
00196         t = t + (t >> 3); /* no content arrived */
00197         if (t > h->predicted_response_limit)
00198             t = h->predicted_response_limit;
00199     }
00200     else if (faceid == s->src) {
00201         t = t - (t >> 7); /* content arrived on expected face */
00202         if (t < 127)
00203             t = 127;
00204     }
00205     s->usec = t;
00206     if (faceid == CCN_NOFACEID)
00207         return;
00208     /* content arrived, so keep track of the arrival face */
00209     if (s->src == CCN_NOFACEID)
00210         s->src = faceid;
00211     else if (s->src != faceid) {
00212         s->osrc = s->src;
00213         s->src = faceid;
00214     }
00215 }
