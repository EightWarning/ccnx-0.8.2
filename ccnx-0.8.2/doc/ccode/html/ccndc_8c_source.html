00077     int res;

00080     if (self == NULL) {
00081         ON_ERROR_EXIT (-1, msg);
00082     }
00083     
00084     self->ccn_handle = ccn_create();
00085     ON_ERROR_EXIT(ccn_connect(self->ccn_handle, NULL), "Unable to connect to local ccnd");
00086     ON_ERROR_EXIT(ccndc_get_ccnd_id(self), "Unable to obtain ID of local ccnd");
00087     
00088     /* Set up an Interest template to indicate scope 1 (Local) */
00089     self->local_scope_template = ccn_charbuf_create();
00090     res = ccnb_element_begin(self->local_scope_template, CCN_DTAG_Interest);
00091     res |= ccnb_element_begin(self->local_scope_template, CCN_DTAG_Name);
00092     res |= ccnb_element_end(self->local_scope_template);        /* </Name> */
00093     res |= ccnb_tagged_putf(self->local_scope_template, CCN_DTAG_Scope, "1");
00094     res |= ccnb_element_end(self->local_scope_template);        /* </Interest> */
00095     ON_ERROR_EXIT(res, msg);
00096     
00097     /* Create a null name */
00098     self->no_name = ccn_charbuf_create();
00099     ON_ERROR_EXIT(ccn_name_init(self->no_name), msg);
00100     
00101     self->lifetime = (~0U) >> 1;
00102     
00103     return self;

00585         newface = ccndc_do_face_action(self, "newface", face);
00586         if (newface == NULL) {
00587             ccndc_warn(__LINE__, "Cannot create/lookup face\n");
00588             goto Cleanup;
00589         }
00590         face->faceid = newface->faceid;
00591         ccn_face_instance_destroy(&newface);
00592         // destroy the old face
00593         newface = ccndc_do_face_action(self, "destroyface", face);
00594         if (newface == NULL) {
00595             ccndc_warn(__LINE__, "Cannot destroy face %d or the face does not exist\n", face->faceid);
00596             goto Cleanup;
00597         }
00598         ccn_face_instance_destroy(&newface);
00599         // recreate the face
00600         newface = ccndc_do_face_action(self, "newface", face);
00601         if (newface == NULL) {
00602             ccndc_warn(__LINE__, "Cannot create/lookup face\n");
00603             goto Cleanup;
00604         }
00605         prefix->faceid = newface->faceid;
00606         ccn_face_instance_destroy(&newface);
00607         // and add the prefix to it
00608         ret_code = ccndc_do_prefix_action(self, "prefixreg", prefix);
00609         if (ret_code < 0) {
00610             ccndc_warn(__LINE__, "Cannot register prefix [%s]\n", cmd_uri);
00611             goto Cleanup;
00612         }
00613     }  
00614     ret_code = 0;
00615 Cleanup:
00616     ccn_face_instance_destroy(&face);
00617     ccn_forwarding_entry_destroy(&prefix);
00618     free(cmd);
00619     return (ret_code);
00620 }

00676     struct ccn_charbuf *uri;
00677     struct ccn_charbuf *uri_auto = NULL;

00679     struct ccn_face_instance *newface;

00687     }

00693     }
00694     
00695     snprintf (port_str, sizeof(port_str), "%d", port);
00696     
00697     /* now process the results */
00698     /* pflhead, lineno=0, "add" "ccnx:/asdfasdf.com/" "tcp|udp", host, portstring, NULL NULL NULL */
00699

00703                                    (~0U) >> 1);

00706                                         self->lifetime);

00709         goto Cleanup;
00710     }

00714     // Second. Destroy the face

00718     if (newface == NULL) {

00732     }
00733     
00734     newface = ccndc_do_face_action(self, "newface", face);
00735     if (newface == NULL) {
00736         ccndc_warn(__LINE__, "Cannot create/lookup face\n");
00737         res = -1;
00738         goto Cleanup;
00739     }

00742     ccn_face_instance_destroy(&newface);

00752                                         self->lifetime);
00753     if (prefix_auto == NULL) {
00754         res = -1;
00755         goto Cleanup;
00756     }
00757 
00758     prefix_auto->faceid = prefix->faceid;
00759     res = ccndc_do_prefix_action(self, "prefixreg", prefix_auto);

00761         ccndc_warn(__LINE__, "Cannot register prefix_auto [%s]\n", ccn_charbuf_as_string(uri_auto));
00762     }

00904         goto ExitOnError;
