00192 {
00193     int configerrors;
00194     int lineno;
00195     
00196     FILE *cfg;
00197     char buf[1024];

00200     int res;
00201     int phase;
00202     int retcode;
00203     int len;
00204     
00205     for (phase = 1; phase >= 0; --phase) {
00206         configerrors = 0;
00207         retcode = 0;
00208         lineno = 0;
00209         cfg = fopen(filename, "r");
00210         if (cfg == NULL)
00211             ccndc_fatal(__LINE__, "%s (%s)\n", strerror(errno), filename);
00212         
00213         while (fgets((char *)buf, sizeof(buf), cfg)) {
00214             lineno++;
00215             len = strlen(buf);
00216             if (buf[0] == '#' || len == 0)
00217                 continue;
00218             if (buf[len - 1] == '\n')
00219                 buf[len - 1] = '\0';
00220             cp = index(buf, '#');
00221             if (cp != NULL)
00222                 *cp = '\0';
00223             
00224             rest_of_the_command = buf;
00225             do {
00226                 cmd = strsep(&rest_of_the_command, " \t");
00227             } while (cmd != NULL && cmd[0] == 0);
00228             
00229             if (cmd == NULL) /* blank line */
00230                 continue;
00231             res = ccndc_dispatch_cmd(ccndc, phase, cmd, rest_of_the_command, -1);
00232             retcode += res;
00233             if (phase == 1 && res < 0) {
00234                 ccndc_warn(__LINE__, "Error: near line %d\n", lineno);
00235                 configerrors++;
00236             }
00237         }
00238         fclose(cfg);
00239         if (configerrors != 0)
00240             return (-configerrors);
00241     }
