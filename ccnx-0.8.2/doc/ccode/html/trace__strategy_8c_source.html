00027 /**
00028  * Append a human-readable rendition of the information in a pit face item
00029  */
00030 static void
00031 format_pfi(struct ccnd_handle *h, struct pit_face_item *p, struct ccn_charbuf *c)
00032 {
00033     struct face *face;
00034     unsigned delta;
00035     
00036     face = ccnd_face_from_faceid(h, p->faceid);
00037     ccn_charbuf_putf(c, " %s%s%s%s%s%s%s%s%u",
00038                      ((p->pfi_flags & CCND_PFI_UPSTREAM) ? "u" :
00039                       (p->pfi_flags & CCND_PFI_DNSTREAM) ? "d" : "?"),
00040                      (p->pfi_flags & (CCND_PFI_PENDING|CCND_PFI_UPENDING)) ? "p" : "",
00041                      (p->pfi_flags & CCND_PFI_UPHUNGRY) ? "h" : "",
00042                      (p->pfi_flags & CCND_PFI_SENDUPST) ? "s" : "",
00043                      (p->pfi_flags & CCND_PFI_ATTENTION) ? "a" : "",
00044                      (p->pfi_flags & CCND_PFI_INACTIVE) ? "q" : "",
00045                      (p->pfi_flags & CCND_PFI_SUPDATA) ? "x" : "",
00046                      (p->pfi_flags & CCND_PFI_DCFACE) ? "c" : "",
00047                      (unsigned)p->faceid);
00048     if (face != NULL) {
00049         if ((p->pfi_flags & CCND_PFI_DNSTREAM) != 0)
00050             ccn_charbuf_putf(c, "-%d", (int)face->pending_interests);
00051         else
00052             ccn_charbuf_putf(c, "+%d", (int)face->outstanding_interests);
00053     }
00054     delta = p->expiry - h->wtnow; /* uses wrapping arithmetic */
00055     if (delta <= 0xffffff)
00056         ccn_charbuf_putf(c, "@%u", delta);
00057 }
